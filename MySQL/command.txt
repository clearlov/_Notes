|~ grant privileges ~|
sql> grant [ PRIVILEGES ] on [ DB ].[ TB ] to [ USERNAME ]@[ HOST =localhost] identified by "[ PASSWD ]"
		[ PRIVILEGES ] all privileges | (select,delete,update,create,drop,rule,insert)
		[ SELECT,DELETE .... ] one or some of all privileges
		<DB> <TB> : you can use * to mean all
		<HOST> :  normally it's localhost
sql> flush privileges;
|- drop user -|
	sql> drop user <USERNAME>@<HOST>
|- Open 3306 for remote ip -|
/**
 * Via Grant Privileges
 * % indicates all ip 
 */
sql> grant all privileges on *.* to root@'%' identified by "[ PASSWD ]"  with grant option;

 * Via Update
 */
sql> use mysql; update user set host='%' where user='root';

	
	
|+ group by +|
|- q: calculate the population of each province -|
	sql> select province, count(*) from tb group by province;
`	sql> select province, count(*) as nu from tb group by province order by nu desc
`   sql> select province, count(*) from tb group by 1 order by 2 desc;
|- q: a field birth(int) in tb, list the staffs' birthdays in each month-|
	sql> select month(birth) as month, count(*) as nu from tb 
	sql> group by month order by nu desc;
`   sql> select month(birth), count(*) from tb group by 1 order by 2 desc;

|+ select multiple tables +|
|- group by -|
	sql> select tb1.date, tb2.profit,
	sql> min(tb3.price) as min,
	sql> max(tb3.price) - min(tb3.price) + 1 as range,
	sql> from tb1, tb2, tb3 
	sql> where tb1.id = tb2.id and tb2.id = tb3.id
	sql> group by tb1.date, tb2.profit
	sql> order by range desc
	|- join -|
	sql> select fld1 from tb1 left join tb2 on (tb1.id = tb2.id)
|+ union and union all +|
|- union : merge the same lines together -|
	sql> (select flds from tb where $where) 
	sql> union
	sql> (select flds from tb where $where)
	sql> order by fld desc limit $limit, $offset
|- union all : keep repeated lines -|
	sql> (select flds from tb where $where order by $order)
	sql> union
	sql> (select flds from tb where $where)
	sql> limit $limit, $offset
